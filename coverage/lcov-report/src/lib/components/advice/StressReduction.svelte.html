	function toggleActive(index) {

	// Function to format description with bold text
	function formatDescription(description) {
		const boldRegex = /\*\*(.*?)\*\*/g;
		return description
			.split('\n')
			.map((line) => {
				return line.replace(boldRegex, '<strong>$1</strong>');
			})
			.join('<br>');
	}
